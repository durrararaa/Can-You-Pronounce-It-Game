    <div class="game-container">
        <div id="start-screen">
            <h1>Pronunciation Quest 🗣️</h1>

            <button class="btn" onclick="startGame()">Start Game</button>

            <div id="word-display" class="word-display">Apple</div>
            
            <button id="mic-btn" class="btn btn-mic" onclick="toggleListening()">🎙️</button>
            
            <div id="feedback" class="feedback info-text">Click the mic and say the word!</div>
            <div id="transcript" class="transcript"></div>
        </div>

        <div id="end-screen" class="hidden">
            <h1> 🎉 Congratulations!</h1>
            <p>You completed all pronunciation tasks perfectly!</p>

    <script>
        // Game Configuration: Add your custom words/phrases here
        const wordList = ["Apple", "Beautiful", "Wednesday", "Comfortable", "Squirrel"];
        
        let currentWordIndex = 0;
        let recognition;
        let isListening = false;

        // Initialize Speech Recognition
        const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
        
        if (SpeechRecognition) {
            recognition = new SpeechRecognition();
            recognition.continuous = false;
            recognition.lang = 'en-US'; // Target Language
            recognition.interimResults = false;
            recognition.maxAlternatives = 1;

            recognition.onstart = () => {
                isListening = true;
                document.getElementById('mic-btn').classList.add('listening');
                updateFeedback("Listening... Speak now!", "info-text");
            };

            recognition.onspeechend = () => {
                stopMicUI();
            };

            recognition.onerror = (event) => {
                stopMicUI();
                updateFeedback("Error occurred: " + event.error, "error-text");
            };

            recognition.onresult = (event) => {
                const speechResult = event.results[0][0].transcript.toLowerCase().trim();
                const targetWord = wordList[currentWordIndex].toLowerCase().trim();
                
                document.getElementById('transcript').innerText = `You said: "${speechResult}"`;

                // Clean punctuation for better matching accuracy
                const cleanResult = speechResult.replace(/[.,\/#!$%\^&\*;:{}=\-_`~()]/g,"");

                if (cleanResult === targetWord) {
                    updateFeedback("Correct! Well done! 🎉", "success-text");
                    document.getElementById('mic-btn').disabled = true;
                    
                    // Move to next word after a short delay
                    setTimeout(() => {
                        currentWordIndex++;
                        if (currentWordIndex < wordList.length) {
                            setupLevel();
                        } else {
                            showScreen('end-screen');
                        }
                    }, 1500);
                } else {
                    updateFeedback("Not quite right. Try again!", "error-text");
                }
            };
        } else {
            alert("Sorry, your browser does not support Speech Recognition. Please try Google Chrome or Safari.");
        }

        // Game Flow Functions
        function startGame() {
            currentWordIndex = 0;
            showScreen('game-screen');
            setupLevel();
        }

        function restartGame() {
            showScreen('start-screen');
        }

        function setupLevel() {
            document.getElementById('mic-btn').disabled = false;
            document.getElementById('word-display').innerText = wordList[currentWordIndex];
            document.getElementById('transcript').innerText = "";
            updateFeedback("Click the mic and say the word!", "info-text");
            
            // Update progress & labels
            document.getElementById('level-indicator').innerText = `Word ${currentWordIndex + 1} of ${wordList.length}`;
            const progressPercent = (currentWordIndex / wordList.length) * 100;
            document.getElementById('progress').style.width = `${progressPercent}%`;
        }

        function toggleListening() {
            if (!recognition) return;
            if (isListening) {
                recognition.stop();
            } else {
                recognition.start();
            }
        }

        function stopMicUI() {
            isListening = false;
            document.getElementById('mic-btn').classList.remove('listening');
        }

        function updateFeedback(text, className) {
            const fb = document.getElementById('feedback');
            fb.innerText = text;
            fb.className = "feedback " + className;
        }

        function showScreen(screenId) {
            document.getElementById('start-screen').classList.add('hidden');
            document.getElementById('game-screen').classList.add('hidden');
            document.getElementById('end-screen').classList.add('hidden');
            
            document.getElementById(screenId).classList.remove('hidden');
        }
    </script>
